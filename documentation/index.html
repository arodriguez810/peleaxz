Iniciar #back to top

Para la autenticación se debe configurar el siguiente archivo:

0-config/security/auth.json


                                        {
                                          "users": {
                                            "model": "my_user", //tabla o vista
                                            "engine": "mysql", //motor de base de datos
                                            "fields": {
                                              "username": "username", //campo username para login
                                              "password": "password", //campo password para login
                                              "id": "id" // id
                                            },
                                            "addFields": { //funciones
                                              "fullName": "capitalize(this.name+space+this.lastname)",
                                              "getID": "this.id"
                                            },
                                            "hash": "md5", // metodo de hash
                                            "expire": "24h", //expiración para la sección
                                            "attempts": 4, //intentos
                                            "unlockTime": 3, //desbloqueo en minutos
                                            "super": [
                                              "angel" //usuario superuser
                                            ]
                                          },
                                          "audit": { //campos de auditoría generales
                                            "insert": {
                                              "user_created": "new SESSION().current().getID()"
                                            },
                                            "update": {
                                              "user_updated": "new SESSION().current().getID()"

                                        }

Carpeta: 3-controllers, los controllers son los que se encargan de generar la ruta automatica para exponer la vista

Controller Normal - #back to top

Estos controllers se utilizan para CRUD's común y corriente por lo tanto deben estar acompañado de su archivo CRUD_controllername 


                                            app.controller("elnombre", function ($scope, $http, $compile) {
                                                elnombre = this; //esta variable es la primera en declarar
                                                //esto es para cargar todas las funcionalidades generales
                                                RUNCONTROLLER("elnombre", elnombre, $scope, $http, $compile);
                                                //esto prepara el metodo para abrir el formulario
                                                elnombre.formulary = function (data, mode, defaultData) {
                                                    if (elnombre !== undefined) {
                                                        //esto es para cargar las funcionalidades de formulario
                                                        RUN_B("elnombre", elnombre, $scope, $http, $compile);
                                                        elnombre.form.readonly = {};
                                                        //esto es para crear el formulario
                                                        elnombre.createForm(data, mode, defaultData);
                                                        //aqui colocamos un vigilador para el valor escrito
                                                        $scope.$watch('elnombre.name', function (value) {
                                                            var rules = [];
                                                            rules.push(VALIDATION.general.required(value));
                                                            VALIDATION.validate(elnombre, "name", rules);
                                                        });

                                                        //En caso de que quieras un vigilador para un objeto o arreglo
                                                        $scope.$watchCollection('elnombre.array', function (value) {

                                                        });
                                                    }
                                                };
                                            });

BASEAPI #back to top

BASEAPI es un conjunto de metodos para realizar todo tipo de consultas, estos metodos utilizan las mismas sintaxis para cualquier tipo de base de datos

BASEAPI LIST - #back to top

Este metodo sirve para traer un listado de registris con las condiciones especificadas

                                        //normal
                                        BASEAPI.list('tabla', {}, function (result) {});

                                        //Where
                                        BASEAPI.list('tabla', {

                                            "where": [
                                                {
                                                    "field": "id",
                                                    "operator": "=",
                                                    "value": 1,
                                                    "connector": "OR"
                                                }
                                            ]
                                        }, function (result) {
                                        });

                                        //Utilizar el caracter "$" para evitar que se encierre entre ''
                                        BASEAPI.list('tabla', {
                                            "where": [
                                                {
                                                    "field": "deleted",
                                                    "operator": "",
                                                    "value": "$Is Null",
                                                }
                                            ]
                                        }, function (result) {
                                        });

                                        //Join y adicionales
                                        BASEAPI.list('tabla', {
                                            limit: 10, //limite por pagina
                                            page: 2, //pagina a traer
                                            orderby: 'desc',
                                            order: 'name', //['id','name'] puede ser un arreglo
                                            join:[
                                                {
                                                    "table": "ms_child", //tabla a unir
                                                    "base": "child", // tabla base
                                                    "field": "id", //campo conexión
                                                    "columns": ["id", "name"] //campos a traer en el join
                                                    //estos campos tomarán el alias ms_child_id y ms_child_id
                                                },
                                                {
                                                    "type": "LEFT", //tipo de join
                                                    "table": "ms_other",
                                                    "base": "other",
                                                    "field": "id",
                                                    "columns": ["id", "description"]
                                                },
                                                {
                                                    "table": "ms_category",
                                                    "base": "category",
                                                    "field": "id",
                                                    "columns": ["id", "name"]
                                                },
                                                {
                                                    "table": "ms_product",
                                                    "base": "product",
                                                    "field": "id",
                                                    "columns": ["id", "name"]
                                                }
                                            ],
                                            "where": [
                                                {
                                                    "field": "id",
                                                    "operator": ">",
                                                    "value": 1,
                                                    "connector": "OR"
                                                }
                                            ]
                                        }, function (result) {
                                        });

                                       //normal
                                       BASEAPI.insertID(
                                        'table',
                                        {campo:1},
                                        "campo",
                                        "1", function (result) {});

BASEAPI updateall - #back to top

Este método sirve para actualizar un registro o varios con los setcampos

                                      BASEAPI.updateall('tabla', {
                                      setcampo:1,
                                      where:[], //consulta el where en el método list
                                      }, function (udata) {});
